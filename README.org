  plotfile = "co2-by-week.png"
  weekday_names = "Mon Tue Wed Thu Fri Sat Sun Mon".split(' ')

  df_inside = pd.read_csv("netatmo_station.csv", sep=';', index_col=1, parse_dates=True)
